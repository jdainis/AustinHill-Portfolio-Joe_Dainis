Presenting data exercise

This data I will be using for this exercise is from a FiveThirtyEight Titled “The Most Common Unisex Names In America: Is Yours One Of Them?”. The data contains the over 900 names given to each sex at least one-third of the time and with a minimum of 100 people. It has the following variables: name: First names from the Social Security Administration total: Total number of living Americans with the name male_share Percentage of people with the name who are male female_share: Percentage of people with the name who are female gap: Gap between male_share and female_share

# Keep top 20 (the graphic shows 1–20)
top20 <- names_df %>% filter(rank <= 20)
# If shares are in 0–100 instead of 0–1, convert to proportions
if (max(top20$male_share, na.rm = TRUE) > 1.5) {
  top20 <- top20 %>%
    mutate(
      male_share = male_share / 100,
      female_share = female_share / 100
    )
}
# Long format for stacked bars
bars <- top20 %>%
  select(rank, name, male_share, female_share) %>%
  pivot_longer(
    cols = c(male_share, female_share),
    names_to = "gender",
    values_to = "share"
  ) %>%
  mutate(
    gender = recode(gender,
      male_share = "Male share",
      female_share = "Female share"
    )
  )
# Order so rank 1 appears at the top
top20 <- top20 %>% mutate(name = factor(name, levels = rev(top20$name)))
bars  <- bars  %>% mutate(name = factor(name, levels = levels(top20$name)))

# Create useful summary statistics (AUTO-CALCULATED)
summary_tbl <- names_df %>%
  filter(rank <= 20) %>%
  mutate(
    dominance = case_when(
      male_share > 0.60 ~ "Mostly male",
      female_share > 0.60 ~ "Mostly female",
      TRUE ~ "Balanced"
    ),
    male_pct = male_share * 100,
    female_pct = female_share * 100
  ) %>%
  select(
    Rank = rank,
    Name = name,
    `People With Name` = total,
    `Male %` = male_pct,
    `Female %` = female_pct,
    Dominance = dominance
  )
